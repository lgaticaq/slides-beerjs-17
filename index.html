docker-compose.yml

version: '2'
services:
  app:
    build: .
    ports:
      - "3000:80"
      - "9229:9229"
    environment:
      MONGODB_URI: mongodb://mongo/myapp
    volumes:
     - .:/opt/app
     - node_modules:/opt/app/node_modules
    depends_on:
      - mongo
    networks:
      - app
  mongo:
    image: mvertes/alpine-mongo
    volumes:
      - mongo:/data/db
    networks:
      - app
volumes:
  mongo:
  node_modules:
networks:
  app:

exit
